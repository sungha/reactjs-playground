1   package kr.sungha.boot.jpa.repository;
2   
3   
4   import java.io.Serializable;
5   import javax.persistence.EntityManager;
6   
7   import org.springframework.data.jpa.repository.support.JpaEntityInformation;
8   import org.springframework.data.jpa.repository.support.JpaRepositoryFactory;
9   import org.springframework.data.repository.core.RepositoryMetadata;
10  import org.springframework.data.repository.core.support.RepositoryComposition;
11  import org.springframework.data.repository.core.support.RepositoryFragment;
12  
13  
14  /**
15   * 기본 리파지토리 빈 생성.
16   *
17   * @param <T> 엔티티 클래스
18   * @param <K> PK 타입
19   */
20  public class BaseRepositoryFactory<T, K extends Serializable> extends JpaRepositoryFactory {
21  
22    private final EntityManager em;
23  
24  
25    public BaseRepositoryFactory(EntityManager em) {
26      super(em);
27      this.em = em;
28    }
29  
30  
31    @Override
32    protected RepositoryComposition.RepositoryFragments getRepositoryFragments(RepositoryMetadata metadata) {
33      RepositoryComposition.RepositoryFragments fragments = super.getRepositoryFragments(metadata);
34  
35      if (BaseRepositoryCustom.class.isAssignableFrom(metadata.getRepositoryInterface())) {
36        JpaEntityInformation<?, Serializable> infomation = getEntityInformation(metadata.getDomainType());
37        Object fragment = getTargetRepositoryViaReflection(BaseRepositoryImpl.class, infomation, em);
38        fragments = fragments.append(RepositoryFragment.implemented(fragment));
39      }
40  
41      return fragments;
42    }
43  
44  }
