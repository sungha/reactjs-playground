20    @QueryDelegate(Timestamp.class)
21    public static BooleanExpression inDatePeriod(DateTimePath<Timestamp> timestamp, Pair<Date, Date> period) {
22      Timestamp first = new Timestamp(DateUtils.truncate(period.getFirst(), Calendar.DAY_OF_MONTH).getTime());
23      Calendar second = Calendar.getInstance();
24      second.setTime(DateUtils.truncate(period.getSecond(), Calendar.DAY_OF_MONTH));
25      second.add(1, Calendar.DAY_OF_MONTH);
26      return timestamp.goe(first).and(timestamp.lt(new Timestamp(second.getTimeInMillis())));
27    }
