1   package kr.sungha.user.service;
2   
3   import kr.sungha.user.entity.Role;
4   import kr.sungha.user.entity.User;
5   
6   import java.util.Map;
7   
8   import lombok.Builder;
9   import lombok.Getter;
10  
11  @Getter
12  @Builder
13  public class OAuthAttributes {
14    private final Map<String, Object> attributes;
15    private final String nameAttributeKey;
16    private final String realname;
17    private final String email;
18  
19    public OAuthAttributes(Map<String, Object> attributes, String nameAttributeKey, String realname, String email) {
20      this.attributes = attributes;
21      this.nameAttributeKey = nameAttributeKey;
22      this.realname = realname;
23      this.email = email;
24    }
25  
26    public static OAuthAttributes of(String registrationId, String userNameAttributeName, Map<String, Object> attributes) {
27      if ("naver".equals(registrationId)) {
28        return ofNaver("id", attributes);
29      }
30  
31      return ofGoogle(userNameAttributeName, attributes);
32    }
33  
34    private static OAuthAttributes ofGoogle(String userNameAttributeName, Map<String, Object> attributes) {
35      return OAuthAttributes.builder()
36          .realname((String) attributes.get("name"))
37          .email((String) attributes.get("email"))
38          .attributes(attributes)
39          .nameAttributeKey(userNameAttributeName)
40          .build();
41    }
42  
43    @SuppressWarnings("unchecked")
44    private static OAuthAttributes ofNaver(String userNameAttributeName, Map<String, Object> attributes) {
45      Map<String, Object> response = (Map<String, Object>) attributes.get("response");
46  
47      return OAuthAttributes.builder()
48          .realname((String) response.get("name"))
49          .email((String) response.get("email"))
50          // .picture((String) response.get("profile_image"))
51          .attributes(response)
52          .nameAttributeKey(userNameAttributeName)
53          .build();
54    }
55  
56    public User toEntity() {
57      return User.builder()
58          .realname(realname)
59          .email(email)
60          .role(Role.GUSET)
61          .build();
62    }
63  }
