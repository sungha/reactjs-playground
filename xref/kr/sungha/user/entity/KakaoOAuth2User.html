13  
14  @Data
15  public class KakaoOAuth2User implements OAuth2User {

19    private Map<String, Object> kakaoAccount;
20    private Map<String, Object> properties;
21  
22    @Override
23    public Map<String, Object> getAttributes() {
24      Map<String, Object> attributes = new HashMap<>();
25      attributes.putAll(kakaoAccount);
26      attributes.putAll(properties);

28    }
29  
30    @Override

33    }
34

38    }
39  }
